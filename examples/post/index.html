.post(url)
Creates an XHR request to the server with the POST verb.

Code example

                server.post("/examples/post/Page.asp")
                .data({ message: "Success", foo:"bar"})
                .success(function (response) {
                    var d = document.getElementById("success");
                    d.innerHTML = response;
                })
                .send();

                server.post("/foo/bar")
                .data({})
                .fail(function (state) {
                    var d = document.getElementById("fail");
                    d.innerHTML = "Fail";
                })
                .send();
